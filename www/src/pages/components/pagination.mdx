# Pagination


##### basic usage

```jsx live=true
<Pagination
  paginationLabel="pagination navigation"
  pageCount={20}
  onPageSelect={() => console.log("page selected")}
/>
```


##### with initial page selected

```jsx live=true
<Pagination
  paginationLabel="pagination navigation"
  pageCount={20}
  currentPage={15}
  onPageSelect={() => console.log("page selected")}
/>
```


##### with max pages displayed

```jsx live=true
<Pagination
  paginationLabel="pagination navigation"
  pageCount={20}
  onPageSelect={() => console.log("page selected")}
  maxPagesDisplayed={9}
/>
```


##### with custom string labels

```jsx live=true
<Pagination
  paginationLabel="pagination navigation"
  pageCount={20}
  onPageSelect={() => console.log("page selected")}
  buttonLabels={{
    previous: 'Anterior',
    next: 'Siguiente',
    page: 'P치gina',
    currentPage: 'P치gina actual',
    pageOfCount: 'de',
  }}
/>
```


##### with custom element labels

```jsx live=true
<Pagination
  paginationLabel="pagination navigation"
  pageCount={20}
  onPageSelect={() => console.log("page selected")}
  buttonLabels={{
    previous: <span>Anterior</span>,
    next: <span>Siguiente</span>,
    page: <span>P치gina</span>,
    currentPage: <span>P치gina actual</span>,
    pageOfCount: <span>de</span>,
  }}
/>
```

  }}
