##### Basic Usage

```jsx live
<Card style={{ width: '18rem' }}>
  <Card.Img variant="top" src="https://source.unsplash.com/400x200/?nature,flower" />
  <Card.Header>
    <Card.Header.Title>Card title</Card.Header.Title>
  </Card.Header>
  <Card.Section>
    <Card.Section.Title>
      Section title
    </Card.Section.Title>
    <Card.Section.Content>
      <Card.Section.Text>
        This is a card section. It can contain anything but usually text, a list, or list of links. Multiple sections have a card divider between them.
      </Card.Section.Text>
    </Card.Section.Content>
    <Card.Section.Actions>
      <Card.Section.Action className="btn btn-outline-primary" href="https://edx.org">
        Action 1
      </Card.Section.Action>
      <Card.Section.Action className="btn btn-tertiary" href="https://edx.org">
        Action 2
      </Card.Section.Action>
    </Card.Section.Actions>
  </Card.Section>
</Card>
```
