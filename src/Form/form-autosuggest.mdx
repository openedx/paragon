Form auto-suggest enables users to manually select or type to find matching options to partial text input.

## Controlled Standalone Usage

```jsx live
() => {
    const [data, setData] = useState({ title: '' });

    return (
        <Form.Autosuggest
            name="color library"
            role="list"
            floatingLabel="This is floating label"
            helpMessage="This is help message"
            ariaLabel="INPUT LABEL"
            value={data.title}
            onChange={
                (value) => setData(prevState => ({ data: { ...prevState.data, org: value } }))
            }
        >
            <Form.AutosuggestOptions type="button" optionsRole="option">Green</Form.AutosuggestOptions>
            <Form.AutosuggestOptions type="button">Red</Form.AutosuggestOptions>
            <Form.AutosuggestOptions type="button">Blue</Form.AutosuggestOptions>
            <Form.AutosuggestOptions type="button">Black</Form.AutosuggestOptions>
        </Form.Autosuggest>
    );
}
```

## With error message

```jsx live
() => {
    const [data, setData] = useState({ title: '' });
    const colors = ['Green', 'Red', 'Blue', 'Black', 'White', 'Purple', 'Orange', 'Grey'];

    return (
        <Form.Autosuggest
            name="color library"
            placeholder="This is placeholder"
            floatingLabel="This is floating label"
            errorMessage="This is error message"
            options={colors}
            value={data.title}
            onChange={
                (value) => setData(prevState => ({ data: { ...prevState.data, org: value } }))
            }
        />
    );
}
```

## Loading state

```jsx live
() => {
    const colors = ['Green', 'Red'];

    return (
        <Form.Autosuggest
            style={{ marginBottom: '14rem' }}
            isLoading
            name="color library"
            placeholder="This is placeholder"
            floatingLabel="This is floating label"
            options={colors}
        />
    );
}
```
